# Rename columns and sort the DataFrame
columns_name = ['sepal.length', 'sepal.width', 'petal.length', 'petal.width', 'Species']
df = pd.read_csv('Iris.csv', header=None, names=columns_name)  
df_sorted = df.sort_values(by='sepal.length', ascending=False)
print("\nSorted dataset (descending by sepal.length):\n", df_sorted.head())

# Slice the DataFrame using iloc and rename a column
sliced_df = df.loc[10:19, ['sepal.length','Species']].copy()
sliced_df.rename(columns={'Species': 'Type'}, inplace=True)
print("\nSliced Data (rows 11 to 20 with selected columns):")
print(sliced_df)
